  58.     public function advancedpostAction() {

  64.                     try {
  65.                         $not = isset ( $all ['NOT'] ) ? $all ['NOT'] : false;
  66.                         $value = isset ( $all ['VALUE'] ) ? $all ['VALUE'] : '';

  68.  
  69.                         $myFilter = Twinmusic_Search_Filter::factory ( $filterType );
  70.                         $myFilter->setNot ( $not );
  71.                         $myFilter->setValue ( $value );
  72.                         $myFilter->setSpecial ( $special );
  73.                         $filterSaved [] = $myFilter;
  74.                     } catch ( Exception $e ) {
  75.                     }
  76.                 }
  77.             }
  78.         }
  79.  
  80.         $engine = new Twinmusic_Search_Engine ( );
  81.         foreach ( $filterSaved as $filter ) {

  83.         }
  84.         $engine->run ();
  85.             
  86.         $this->_forward ( 'showresult', 'search', null, array('page' => 1) );
  87.     }
