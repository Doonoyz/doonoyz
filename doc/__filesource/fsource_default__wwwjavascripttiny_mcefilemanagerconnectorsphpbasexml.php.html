  21. function SetXmlHeaders()
  22. {
  23.     // Prevent the browser from caching the result.
  24.     // Date in the past
  25.     header('Expires: Mon, 26 Jul 1997 05:00:00 GMT') ;
  26.     // always modified
  27.     header('Last-Modified: ' . gmdate('D, d M Y H:i:s') . ' GMT') ;

  31.     // HTTP/1.0
  32.     header('Pragma: no-cache') ;
  33.
